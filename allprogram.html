      }
      g.fillPolygon(xPoints, yPoints, 10);

import java.awt.Graphics;
import java.awt.Polygon;

public class FlagDrawing
{
   
   // Method to draw a five-pointed star centered at (centerx, centery) with given radius
   public static void paintstart(int centerx, int centery, int radius, Graphics g)
   {
      // Calculate the coordinates of the five points of the star
      int[] xPoints = new int[10];
      int[] yPoints = new int[10];
      
      double angle = Math.PI / 2; // Start angle at the top point of the star
      double angleIncrement = 2 * Math.PI / 5; // Angle increment between each point of the star
      
      for (int i = 0; i < 5; i++)
      {
         // Outer points of the star
         xPoints[2 * i] = (int) (centerx + radius * Math.cos(angle));
         yPoints[2 * i] = (int) (centery - radius * Math.sin(angle));
         
         // Inner points of the star (half the radius for inner points)
         xPoints[2 * i + 1] = (int) (centerx + radius / 2 * Math.cos(angle + angleIncrement / 2));
         yPoints[2 * i + 1] = (int) (centery - radius / 2 * Math.sin(angle + angleIncrement / 2));
         
         angle += angleIncrement;
      }
      
      // Create a polygon with the calculated points
      Polygon star = new Polygon(xPoints, yPoints, 10);
      
      // Draw the star on the Graphics object
      g.drawPolygon(star);
   }
}

   public static Shape createPentacle(double sx, double sy, double radius, double theta)
   {
      final double arc = Math.PI / 5;
      final double rad = Math.sin(Math.PI / 10) / Math.sin(3 * Math.PI / 10);
      GeneralPath path = new GeneralPath();
      path.moveTo(1, 0);
      for (int i = 0; i < 5; i++)
      {
         path.lineTo(rad * Math.cos((1 + 2 * i) * arc), rad * Math.sin((1 + 2 * i) * arc));
         path.lineTo(Math.cos(2 * (i + 1) * arc), Math.sin(2 * (i + 1) * arc));
      }
      path.closePath();
      AffineTransform atf = AffineTransform.getScaleInstance(radius, radius);
      atf.translate(sx / radius, sy / radius);
      atf.rotate(theta);
      return atf.createTransformedShape(path);
   }
   
   private int width, height;
   private double maxR = 0.15, minR = 0.05;
   private double maxX = 0.25, maxY = 0.25;
   private double[] minX =
   {
      0.50, 0.60, 0.60, 0.50
   };
   private double[] minY =
   {
      0.10, 0.20, 0.35, 0.45
   };
   
   public FiveStarFlag(int width)
   {
      this.width = width / 3 * 3;
      this.height = width / 3 * 2;
      setPreferredSize(new Dimension(this.width, this.height));
   }
   
   @Override
   protected void paintComponent(Graphics g)
   {
      Graphics2D graphics2D = (Graphics2D) g;
      
      graphics2D.setPaint(Color.RED);
      graphics2D.fillRect(0, 0, width, height);
      
      double ox = height * maxX, oy = height * maxY;
      graphics2D.setPaint(Color.YELLOW);
      graphics2D.fill(createPentacle(ox, oy, height * maxR, -Math.PI / 2));
      
      for (int i = 0; i < 4; i++)
      {
         double sx = minX[i] * height, sy = minY[i] * height;
         double theta = Math.atan2(oy - sy, ox - sx);
         graphics2D.fill(createPentacle(sx, sy, height * minR, theta));
      }
   }
}

import java.io.File;

public class DataStatistics
{
   public static void main(String[] args)
   {
      try
      {
         Scanner scanner = new Scanner(new File(args[0]));
         
         int n = 0;
         int row = 0, col = 0;
         float sum;
         String grades = "FDCBA";
         String[] headings = new String[9];
         float[][] nums = new float[48][];
         float[] means = new float[9];
         float[] stdvs = new float[9];
         
         while (scanner.hasNext())
         {
            if (col == 0)
            {
               nums[row] = new float[9];
            }
            if (row == 0 || col == 1)
            {
               if (col == 1)
               {
                  nums[row][col] = grades.indexOf(scanner.next());
               }
               else
               {
                  headings[col] = scanner.next();
               }
            }
            else
            {
               nums[row][col] = scanner.nextFloat();
            }
            col++;
            if (col == 9)
            {
               col = 0;
               row++;
            }
         }
         scanner.close();
         n = row;
         
         // Calculate means
         for (int i = 0; i < 9; i++)
         {
            sum = 0;
            for (int j = 0; j < n; j++)
            {
               sum += nums[j][i];
            }
            means[i] = sum / n;
         }
         
         // Calculate standard deviations
         for (int i = 0; i < 9; i++)
         {
            sum = 0;
            for (int j = 0; j < n; j++)
            {
               sum += (nums[j][i] - means[i]) * (nums[j][i] - means[i]);
            }
            stdvs[i] = (float) Math.sqrt(sum / (n - 1));
         }
         
         // Output results in the specified format
         System.out.println("Mean of id: "+ means[0]);
         System.out.println("Mean of Grade: "+ means[1]);
         System.out.println("Mean of Score: "+ means[2]);
         System.out.println("Mean of Absence: "+ means[3]);
         System.out.println("Mean of Time: "+ means[4]);
         System.out.println("Mean of Start: "+ means[5]);
         System.out.println("Mean of Sub: "+ means[6]);
         System.out.println("Mean of Quiz: "+ means[7]);
         System.out.println("Mean of Order: "+ means[8]);
         System.out.println("Stdv of id: "+ stdvs[0]);
         System.out.println("Stdv of Grade: "+ stdvs[1]);
         System.out.println("Stdv of Score: "+ stdvs[2]);
         System.out.println("Stdv of Absence: "+ stdvs[3]);

public class SimpleHTTPServer
{
   private ServerSocket serverSocket;
   private final int port;
   public SimpleHTTPServer(int port)
   {
      this.port = port;
   }
   public void start()
   {
      try
      {
         serverSocket= new ServerSocket(port);
         System.out.println("Server started on port "+ port);
         while(true)
         {
            Socket clientSocket=serverSocket.accept();
            Thread thread = new Thread(()->handleClient(clientSocket));
            thread.start();
         }
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   private void handleClient(Socket clientSocket)
   {
      try
      {
         BufferedReader in=new BufferedReader(new InputStreamReader(clientSocket.getInputStream()));
         DataOutputStream out= new DataOutputStream(clientSocket.getOutputStream());
         // Read the request headers
         Map headers = readHeaders(in);
         System.out.println(headers);
         // Check ifit's a multipart/form-data request
         // Send response
         String response="HTTP/1.1 200 OK\r\n"+"Content-Type:text/plain\r\n"+"\r\n "+"request received";
         out.writeBytes(response);
         in.close();
         out.close();
         clientSocket.close();
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   private Map readHeaders(BufferedReader in) throws IOException
   {
      Map headers=new HashMap<>();
      String line;
      while (!(line =in.readLine()).isEmpty())
      {
         String[] parts = line.split(":",2);
         if(parts.length==2)
         {
            headers.put(parts[0].trim(), parts [1].trim());
         }
      }
      return headers;
   }

   {
      SimpleHTTPServer server=new SimpleHTTPServer(8080);
      server.start();
   }
